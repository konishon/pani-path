from vcube.engine import VCubeProcessor

# Setup parameters
start_date = "2014-01-01"
end_date = "2025-01-01"
cloud_cover = 10
formula = "(band2-band1)/(band2+band1)"  # NDVI
band1 = "red"
band2 = "nir"
operation = "median"
timeseries = True
workers = 10
smart_filter = False

processor = VCubeProcessor(
    bbox,
    start_date,
    end_date,
    cloud_cover,
    formula,
    band1,
    band2,
    operation,
    timeseries,
    output_dir,
    workers=workers,
    smart_filter=smart_filter
)

processor.compute()
